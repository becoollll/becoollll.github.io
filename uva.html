12019 - Doom's Day Algorithm

    int month[12] = {31,28,31,30,31,30,31,31,30,31,30,31};

    cin >> c;
    while(c--){
        string ans = "";
        int a = 0, mon, day;
        cin >> mon >> day;
        if(mon!=1){
            for(int i = 0; i <; mon-1; i++){

11498 - Division of Nlogonia

            
#include<iostream>
using namespace std;
                    
int main(){
    while(1){
        int k;
        cin >> k;
        if(k == 0){
            break;
        }
        int n, m;//參考
        cin >> n >> m;
        while(k--){
            int x, y;
            cin >> x >> y;
            if(x == n || y == m){
                cout ≪ "divisa" ≪ endl;
            }
            if(x > n && y > m){
                cout ≪ "NE" ≪ endl;
            }
            if(x > n && y < m){
                cout ≪ "SE" ≪ endl;
            }
            if(x < n && y > m){
                cout ≪ "NO" ≪ endl;
            }
            if(x < n && y < m){
                cout ≪ "SO" ≪ endl;
            }
        }
    }
    return 0;
}

11455 - Behold my quadrangle

            
#include <iostream>
#include <vector>
#include <lgorithm>
using namespace std;

int main(){
    int c;
    cin >> c;
    while(c--){
        int kk[4];
        for(int i = 0; i < 4; i++){
            cin >> kk[i];
        }
        sort(kk,kk+4);
        if(kk[0]==kk[3]){
            cout ≪ "square" ≪ endl;
        }
        else if(kk[0]==kk[1] && kk[2]==kk[3]){
            cout ≪ "rectangle" ≪ endl;
        }
        else if(kk[0]+kk[1]+kk[2] >= kk[3]){
            cout ≪ "quadrangle" ≪ endl;
        }
        else{
            cout ≪ "banana" ≪ endl;
        }
    }
}
