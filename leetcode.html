1. Two Sum (Easy)

Runtime: 880 ms
Memory Usage: 10.1 MB

            
class Solution {
    public:
    vector<int> twoSum(vector<int>& nums, int target){
        vector<int> a,ans;
        for(int i = 0; i < nums.size(); i++){
            for(int j = 0; j < nums.size(); j++){
                if(nums[i] + nums[j] == target){
                    if(i != j){
                        a.push_back(i);
                        a.push_back(j);
                        break;
                    }
                }
            }
        }
        for(int i = 0; i < 2; i++){
            ans.push_back(a[i]);
        }
        return ans;
    }
};

9. Palindrome Number (Easy)

Runtime: 0 ms
Memory Usage: 5.8 MB

            
class Solution {
    public:
    bool isPalindrome(int x){
        long long unsigned int sum = 0, n = x;
        if(n < 0){
            return false;
        }
        while(x > 0){
            int r = x % 10;
            sum = (sum * 10 + r);
            x /= 10;
        }              
        return (sum == n);
    }
};

14. Longest Common Prefix (Easy)

Runtime: 0 ms
Memory Usage: 9 MB

            
class Solution {
    public:
    string longestCommonPrefix(vector<string>& strs) {
        string ans = strs[0];
        for(int i = 0; i < strs.size(); i++){
            for(int j = 0; j < ans.size(); j++){
                if(strs[i][j] != ans[j]){
                    ans.erase(j);
                }
            }
        }
        return ans;
    }
};

21. Merge Two Sorted Lists (Easy)

Runtime: 12 ms
Memory Usage: 14.8 MB

            
class Solution {
    public:
    ListNode *mergeTwoLists(ListNode *l1, ListNode *l2) {
        if(l1 == NULL){
            return l2;
        }
        else if(l2 == NULL){
            return l1;
        }
        ListNode *head = NULL;   //for return 
        if(l1->val <= l2->val){
            head = l1;
            l1 = l1->next;
        }
        else{
            head = l2;
            l2 = l2->next;
        }      
        ListNode *temp = head; 
        while(l1 != NULL && l2 != NULL){
            if(l1->val <= l2->val){
                temp->next = l1;
                l1 = l1->next;
            }
            else{
                temp->next = l2;
                l2 = l2->next;
            }
            temp = temp->next;
        }
        if(l1 != NULL){
            temp->next = l1;
        }
        else{
            temp->next = l2;
        }
        return head;
    }
};

26. Remove Duplicates from Sorted Array (Easy)

Runtime: 20 ms
Memory Usage: 8.8 MB

            
class Solution {
    int removeDuplicates(int* nums, int numsSize) {
    if (numsSize==0)
        return 0;

    int i=0;
    for (int j=0;j<numsSize;j++)
    {
        if (nums[j]!=nums[i]){   //找到沒有重複的值
            i++;
            nums[i] = nums[j];   //把位於nums[j](為無重複的值)移動到i++的位置
        }
    }
    return i + 1;   //+1是因為陣列從0開始

    }
};

58. Length of Last Word (Easy)

Memory Usage: 6.6 MB

            
class Solution {
    public:
    int lengthOfLastWord(string s) {
            int c = 0;
            for(int i = s.length()-1; i >= 0; i--){
                if(s[i] == ' ' && c > 0){
                    return c;
                }
                if(s[i] != ' '){
                    c++;
                }
            }
            return c;
    }
};

        for(int i = digits.size()-1; i >= 0; i--){
            if(digits[i] == 9){
                digits[i] = 0;
            }
            else{
                digits[i]++;
                return digits;
            }
        }
        if(digits[0] == 0){
            digits.insert(digits.begin(), 1);
        }
        return digits;
    }

70. Climbing Stairs (Easy)

Runtime: 0 ms
Memory Usage: 5.9 MB

            
class Solution {
    public:
    int climbStairs(int n){
        if(n <= 0){
            return -1;
        }
        if(n == 1){
            return 1;
        }
        if(n == 2){
            return 2;
        }
        int vt[100];
        vt[0]=1;
        vt[1]=2;
        for(int i = 2; i < n; i++){
            vt[i] = vt[i-1] + vt[i-2];
        }
        return vt[n-1];
    }
};

class Solution {

    ListNode* deleteDuplicates(ListNode* head) {
        ListNode *temp = head;
        while(temp != NULL){
            if(temp->next != NULL && temp->val == temp->next->val){
                temp->next = temp->next->next;
            }
            else{
                temp = temp->next;
            }
        }
        return head;
    }
};

100. Same Tree (Easy)

Memory Usage: 10 MB

            
class Solution {
    public:
    bool isSameTree(TreeNode* p, TreeNode* q) {
        if(p == NULL && q == NULL){
            return true;
        }
        if(p == NULL && q != NULL){
            return false;
        }
        if(p != NULL && q == NULL){
            return false;
        }
        return (p->val == q->val && isSameTree(p->right, q->right) && isSameTree(p->left, q->left));
    }
};

118. Pascal's Triangle (Easy)

Runtime: 0 ms
Memory Usage: 6.6 MB

            
class Solution {
    public:
    vector<vector<int>> generate(int numRows) {

        vector<vector<int>> ans;

        if(numRows == 1){
            return {{1}};
        }
        if(numRows == 2){
            return {{1}, {1, 1}};
        }
        for(int i = 0; i < numRows; i++){
            vector<int> temp;
            for(int j = 0; j <= i; j++){
                if(i != 0 && j == 0){   //[1][0],[2][0]...
                    temp.push_back(1);
                }
                if(i != 0 && i != 1 && i != j && j != 0){   
                    temp.push_back(ans[i-1][j-1] + ans[i-1][j]);
                }
                if(i == j){
                    temp.push_back(1);
                    ans.push_back(temp);
                }
            }
        }
        return ans;
    }
};

class Solution {
    public:
    vector<int> getRow(int rowIndex) {
        vector<vector> ans;
        if(rowIndex == 0){
            return {1};
        }
        if(rowIndex == 1){
            return {1, 1};
        }
        for(int i = 0; i < rowIndex+1; i++){
            vector<int> temp;
            for(int j = 0; j <= i; j++){
                if(i != 0 && j == 0){   //[1][0],[2][0]...   
                   temp.push_back(1);
                }
                if(i != 0 && i != 1 && i != j && j != 0){
                    temp.push_back(ans[i-1][j-1] + ans[i-1][j]);
                }
                if(i == j){
                    temp.push_back(1);
                    ans.push_back(temp);
                }
            }
        }
        vector<int> ret;
        for(int i = 0; i <= rowIndex; i++){
            ret.push_back(ans[rowIndex][i]);
        }
        return ret;
    }
};

136. Single Number (Easy)

Runtime: 16 ms
Memory Usage: 16.9 MB

            
class Solution {
    public:
    int singleNumber(vector<int>& nums) {
        int ans = 0;
        if(nums.size() == 1){
            return nums[0];
        }
        sort(nums.begin(), nums.end()); // 1,1,2,2,4
        for(int i = 1; i < nums.size()-1; i++){
            if(nums[i] != nums[i+1] && nums[i] != nums[i-1]){
                ans = nums[i];
            }
        }
        if(nums[0] != nums[1]){
            ans = nums[0];
        }
        else if(nums[nums.size()-2] != nums[nums.size()-1]){
            ans = nums[nums.size()-1];
        }
        return ans;
    }
};

172. Factorial Trailing Zeroes (Easy)

Runtime: 0 ms
Memory Usage: 5.9 MB

            
class Solution {
    public:
    int trailingZeroes(int n) {
        int ans = 0;
        if(n == 0){
            return 0;
        }
        while(n > 0){   //階乘的規律:https://wywu.pixnet.net/blog/post/22014250
            n /= 5;
            ans += n;
        }
        return ans;
    }
};
